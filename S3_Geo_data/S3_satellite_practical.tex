\begin{frame}{Nighttime Lights as an Economic Indicator}
\begin{itemize}
  \item \textbf{Why nighttime lights?}
  \vspace{0.2cm}
  \begin{itemize}
    \item Light intensity at night is a strong proxy for economic activity
    \vspace{0.1cm}
    \item Publicly available, globally consistent, fine spatial resolution
  \end{itemize}
  \vspace{0.3cm}
  \item \textbf{Key reference:}
  \vspace{0.1cm}
  \begin{itemize}
    \item Henderson, Storeygard, and Weil (2012),
          ``\textbf{Measuring Economic Growth from Outer Space}'', AER
    \vspace{0.1cm}
    \item Nighttime lights predict GDP growth at both national and
          sub-national levels
  \end{itemize}
  \vspace{0.3cm}
  \item \textbf{Advantages over official statistics:}
  \vspace{0.1cm}
  \begin{itemize}
    \item Available for regions where official GDP data are poor or absent
    \vspace{0.1cm}
    \item Captures \emph{within-country} spatial variation at fine scale
    \vspace{0.1cm}
    \item Consistent over time and across countries
  \end{itemize}
\end{itemize}
\end{frame}


\begin{frame}{Two Generations of Nighttime Lights Data}
\begin{center}
\small
\begin{tabular}{lcc}
  \toprule
  & \textbf{DMSP-OLS} & \textbf{NASA Black Marble (VIIRS)} \\
  \midrule
  Sensor        & DMSP / OLS      & Suomi-NPP VIIRS DNB \\
  Coverage      & 1992–2013       & 2012–present \\
  Resolution    & $\sim$1 km      & 500 m \\
  Radiometric   & 6-bit (0–63 DN) & 14-bit, calibrated nW/sr/cm$^2$ \\
  Saturation    & Yes (urban)     & No \\
  Moonlight corr.& No             & Yes \\
  Cloud mask    & Partial         & Yes \\
  \bottomrule
\end{tabular}
\end{center}
\vspace{0.2cm}
\begin{itemize}
  \item \textbf{Problem:} DMSP ends in 2013; VIIRS starts in 2012
        --- the two series are \emph{not} directly comparable
  \vspace{0.15cm}
  \item \textbf{Solution:} use a \textbf{harmonized} dataset that
        intercalibrates both sensors onto a common scale
\end{itemize}
\end{frame}

\begin{frame}{Harmonized DMSP-OLS / VIIRS Dataset}
\begin{itemize}
  \item \textbf{Data we use:} Harmonized NTL on the DMSP scale
        (``calDMSP'')
  \vspace{0.2cm}
  \begin{itemize}
    \item Li et al.\ (2020), \textit{Remote Sensing of Environment}
    \vspace{0.1cm}
    \item DMSP-OLS data inter-calibrated across satellites;
          post-2013 VIIRS data converted to DMSP DN scale
    \vspace{0.1cm}
    \item Coverage: 1992 to 2024, global, $\sim$1 km resolution
  \end{itemize}
  \vspace{0.3cm}
  \item \textbf{File naming:}
  \vspace{0.1cm}
  \begin{itemize}
    \item \texttt{Harmonized\_DN\_NTL\_\{year\}\_calDMSP.tif}
    \vspace{0.1cm}
    \item DN range: 0 to 63 (same DMSP scale for all years)
    \vspace{0.1cm}
    \item One GeoTIFF per year, global extent
  \end{itemize}
\end{itemize}
\end{frame}

\begin{frame}{Harmonized DMSP-OLS / VIIRS Dataset}
\begin{itemize}
  \item \textbf{What is GeoTIFF?}
  \vspace{0.15cm}
  \begin{itemize}
    \item A standard raster image format (\texttt{.tif}) with
          \emph{embedded geographic metadata}
    \vspace{0.1cm}
    \item Each pixel carries a real-world coordinate
          (latitude / longitude) and a map projection,
          so the file is self-describing: no separate
          shapefile or header needed
    \vspace{0.1cm}
    \item Every pixel here stores one DN value (0 to 63)
          representing the brightness of a
          ${\sim}$1 km $\times$ 1 km ground cell
    \vspace{0.1cm}
    \item Readable by Python (\texttt{rasterio},
          \texttt{gdal}), R (\texttt{terra}, \texttt{stars}),
          QGIS, ArcGIS, etc.
  \end{itemize}
\end{itemize}
\end{frame}

\begin{frame}{What is a Raster?}
\begin{itemize}
  \item A \textbf{raster} is a grid of regularly spaced cells (pixels),
        each storing a numeric value
  \vspace{0.3cm}
  \item Each cell covers a fixed geographic area (the \textbf{resolution}),
        e.g.\ $\sim$1 km $\times$ 1 km
  \vspace{0.3cm}
  \item A raster is fully described by:
  \vspace{0.15cm}
  \begin{center}
  \small
  \begin{tabular}{ll}
    \toprule
    \textbf{Property} & \textbf{Example (our data)} \\
    \midrule
    Extent   & Global: $-180^\circ$ to $180^\circ$, $-90^\circ$ to $90^\circ$ \\
    Resolution & $\approx 0.0083^\circ$ ($\approx 926$ m at equator) \\
    CRS      & WGS84 (EPSG:4326) \\
    Values   & Digital Number (DN), 0–63 \\
    \bottomrule
  \end{tabular}
  \end{center}
  \vspace{0.3cm}
  \item In R, the \texttt{terra} package represents rasters as
        \texttt{SpatRaster} objects
  \vspace{0.15cm}
  \begin{itemize}
    \item Lazy loading: \texttt{rast()} reads only metadata;
          pixel values are loaded when needed $\Rightarrow$ memory efficient
  \end{itemize}
\end{itemize}
\end{frame}

\begin{frame}{Why CRS Matters}
\begin{itemize}
  \item Every spatial dataset --- raster or vector --- is referenced
        to a \textbf{coordinate reference system (CRS)}
  \vspace{0.3cm}
  \item The CRS defines how coordinates $(x, y)$ map to locations
        on the Earth's surface
  \vspace{0.3cm}
  \item \textbf{Common mistake:} overlaying two layers with
        \emph{different} CRS
  \vspace{0.15cm}
  \begin{itemize}
    \item Polygons will not align with the raster grid
    \vspace{0.1cm}
    \item Extraction results will be wrong or empty
  \end{itemize}
  \vspace{0.3cm}
  \item \textbf{Rule:} always check and harmonize CRS before any
        spatial operation
  \vspace{0.15cm}
  \begin{itemize}
    \item Check: \texttt{terra::crs(r)} / \texttt{sf::st\_crs(shp)}
    \vspace{0.1cm}
    \item Transform vector: \texttt{sf::st\_transform(shp, crs = 4326)}
    \vspace{0.1cm}
    \item Reproject raster: \texttt{terra::project(r, "EPSG:4326")}
  \end{itemize}
\end{itemize}
\end{frame}


\begin{frame}{Two Types of Coordinate Systems}
\begin{itemize}
  \item \textbf{Geographic Coordinate System (GCS)}
  \vspace{0.15cm}
  \begin{itemize}
    \item Measures location in \textbf{degrees}: longitude / latitude
    \vspace{0.1cm}
    \item Three-dimensional angular system referenced to a datum
    \vspace{0.1cm}
    \item Example: WGS84 --- longitude $\in [-180^\circ, 180^\circ]$,
          latitude $\in [-90^\circ, 90^\circ]$
    \vspace{0.1cm}
    \item \textbf{Taipei:} $24.57^\circ$ N, $121.32^\circ$ E
  \end{itemize}
  \vspace{0.35cm}
  \item \textbf{Projected Coordinate System (PCS)}
  \vspace{0.15cm}
  \begin{itemize}
    \item Flattens the globe onto a 2D plane; units in \textbf{meters}
    \vspace{0.1cm}
    \item Needed for accurate distance and area calculations
    \vspace{0.1cm}
    \item Example: TWD97 (Taiwan Datum 1997, EPSG:3826) ---
          used by Taiwan government shapefiles
    \vspace{0.1cm}
    \item All projections introduce some distortion --- choose one
          appropriate for your region
  \end{itemize}
\end{itemize}
\end{frame}


\begin{frame}{EPSG Codes}
\begin{itemize}
  \item \textbf{EPSG} maintains a registry of standardized CRS codes
  \vspace{0.3cm}
  \begin{center}
  \small
  \begin{tabular}{lll}
    \toprule
    \textbf{EPSG} & \textbf{Name} & \textbf{Usage} \\
    \midrule
    4326 & WGS84         & Global; GPS, most satellite data \\
    3826 & TWD97 / TM2   & Taiwan government shapefiles \\
    3857 & Web Mercator  & Web maps (Google Maps, OSM) \\
    32651 & UTM Zone 51N & East Asia, metric distances \\
    \bottomrule
  \end{tabular}
  \end{center}
  \vspace{0.3cm}
  \item \textbf{In this practical:}
  \vspace{0.15cm}
  \begin{itemize}
    \item NTL GeoTIFF $\Rightarrow$ WGS84 (EPSG:4326)
    \vspace{0.1cm}
    \item GADM shapefile via \texttt{geodata::gadm()}
          $\Rightarrow$ already WGS84
    \vspace{0.1cm}
    \item Taiwan gov.\ shapefile $\Rightarrow$ TWD97 (EPSG:3826);
          must \texttt{st\_transform} to 4326 before use
  \end{itemize}
\end{itemize}
\end{frame}


\begin{frame}[fragile]{CRS in R: Checking and Transforming}
\begin{lstlisting}
# Check raster CRS
terra::crs(tif_1992, describe = TRUE)
# -> name: "WGS 84", code: "4326"

# Check vector CRS
sf::st_crs(tw_sf)$epsg   # -> 4326

# Transform vector to WGS84 (required if using
# Taiwan gov. shapefile, which is EPSG:3826)
tw_sf <- sf::st_transform(tw_sf, crs = 4326)

# Reproject raster (if CRS mismatch)
# r_wgs84 <- terra::project(r, "EPSG:4326")
\end{lstlisting}
\begin{itemize}
  \item \texttt{terra::vect()} converts \texttt{sf} $\to$ \texttt{SpatVector}
        for raster operations; both must share the same CRS
\end{itemize}
\end{frame}

\begin{frame}{Where to Get Taiwan Boundaries}

          --- 鄉鎮市區界線 (township / district boundaries,
          $\approx$368 units)
    \vspace{0.1cm}
    \item National Land Surveying and Mapping Center:
          \href{https://maps.nlsc.gov.tw}{\texttt{maps.nlsc.gov.tw}}
    \vspace{0.1cm}
    \item CRS: TWD97 (EPSG:3826) --- must convert to WGS84 first
    \vspace{0.1cm}
    \item Read in R: \texttt{sf::st\_read("TOWN\_MOI.shp")} then

\begin{frame}[fragile]{Step 3: Crop and Mask to Taiwan}
\begin{lstlisting}
# Two-step spatial subsetting
#  crop()  : rectangular clip to bounding box (fast)
#  mask()  : set pixels outside polygons to NA

ntl_1992 <- tif_1992 |>
  terra::crop(tw_vect) |>
  terra::mask(tw_vect)

ntl_2024 <- tif_2024 |>
  terra::crop(tw_vect) |>
  terra::mask(tw_vect)

names(ntl_1992) <- "ntl_1992"
names(ntl_2024) <- "ntl_2024"
\end{lstlisting}
\begin{itemize}
  \item Why two steps? \texttt{crop()} reduces the global raster
        to Taiwan's bounding box first; \texttt{mask()} then removes
        ocean pixels --- doing \texttt{mask()} on the full global
        raster would be very slow
\end{itemize}
\end{frame}


\begin{frame}[fragile]{Step 4: Visualize --- Raster Maps}
\begin{lstlisting}
# Shared upper limit for comparable color scales
val_max <- max(
  max(terra::values(ntl_1992, na.rm = TRUE)),
  max(terra::values(ntl_2024, na.rm = TRUE)))

# Map function: white (DN=0) -> red (high DN)
make_ntl_map <- function(r, title_text,
                          tw_sf, val_max) {
  ggplot() +
    geom_spatraster(data = r) +
    geom_sf(data = tw_sf, fill = NA,
            color = "gray60", linewidth = 0.15) +
    scale_fill_gradient(
      low = "white", high = "firebrick3",
      limits = c(0, val_max),
      oob = scales::squish,
      na.value = "transparent", name = "DN") +
    coord_sf(expand = FALSE) + theme_void()
}

p_1992 <- make_ntl_map(ntl_1992,
  "Taiwan NTL 1992", tw_sf, val_max)
p_2024 <- make_ntl_map(ntl_2024,
  "Taiwan NTL 2024", tw_sf, val_max)

# Side-by-side
p_compare <- p_1992 + p_2024 +
  plot_layout(guides = "collect")
ggsave("ntl_taiwan_compare.png",
       p_compare, width = 11, height = 6, dpi = 300)
\end{lstlisting}
\begin{itemize}
  \item \texttt{limits = c(0, val\_max)} is the same in both panels
        $\Rightarrow$ color scale is directly comparable across years
\end{itemize}
\end{frame}


\begin{frame}[fragile]{Step 4 (cont.): Change Map}

    na.value = "transparent", name = "Change (DN)")
\end{lstlisting}

  \item Blue = declined; Red = increased brightness (2024 vs.\ 1992)

\begin{frame}[fragile]{Step 5: Extract County-Level Statistics}
\begin{lstlisting}
# terra::extract(): assign each pixel to a county polygon
# fun = NULL -> returns all pixel values per polygon
extract_stats <- function(r, tw_vect) {
  ex <- terra::extract(r, tw_vect,
                        fun = NULL, na.rm = TRUE)
  ex |>
    rename(value = all_of(names(r))) |>
    group_by(ID) |>
    summarise(
      ntl_mean   = mean(value,   na.rm = TRUE),
      ntl_sum    = sum(value,    na.rm = TRUE),
      ntl_sd     = sd(value,     na.rm = TRUE),
      ntl_median = median(value, na.rm = TRUE),
      n_pixels   = n(),
      n_valid    = sum(!is.na(value)),
      .groups = "drop")
}

stats_1992 <- extract_stats(ntl_1992, tw_vect) |>
  rename_with(~ paste0(., "_1992"), -ID)
stats_2024 <- extract_stats(ntl_2024, tw_vect) |>
  rename_with(~ paste0(., "_2024"), -ID)
\end{lstlisting}
\end{frame}


\begin{frame}[fragile]{Step 5 (cont.): Join and Compute Changes}
\begin{lstlisting}
# County metadata from shapefile
# NAME_2 = county name (22 counties/cities)
tw_meta <- tw_sf |>
  sf::st_drop_geometry() |>
  mutate(ID = row_number()) |>
  select(ID, province = NAME_1, county = NAME_2)

# Wide-format: one row per county
tw_ntl <- tw_meta |>
  left_join(stats_1992, by = "ID") |>
  left_join(stats_2024, by = "ID") |>
  select(-ID) |>
  mutate(
    ntl_change_mean = ntl_mean_2024 - ntl_mean_1992,
    ntl_pct_change  =
      (ntl_mean_2024 - ntl_mean_1992) /
      (ntl_mean_1992 + 1) * 100)
\end{lstlisting}
\end{frame}
