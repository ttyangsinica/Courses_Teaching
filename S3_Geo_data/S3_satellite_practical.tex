\begin{frame}{What is NASA Black Marble?}
\begin{itemize}
  \item \textbf{NASA Black Marble} is a NASA Earth Science Data Systems (ESDS)
        project that provides daily, monthly, and yearly global nighttime lights
        using the \textbf{VIIRS Day-Night Band (DNB)} sensor
  \vspace{0.3cm}
  \item Sensor: Suomi-NPP and NOAA-20 satellites
  \vspace{0.3cm}
  \item \textbf{Data products:}
  \vspace{0.1cm}
  \begin{center}
  \small
  \begin{tabular}{llll}
    \toprule
    \textbf{Product} & \textbf{Frequency} & \textbf{Resolution} & \textbf{Coverage} \\
    \midrule
    VNP46A1 & Daily     & 500 m & 2012--present \\
    VNP46A2 & Daily (gap-filled) & 500 m & 2012--present \\
    VNP46A3 & Monthly   & 500 m & 2012--present \\
    VNP46A4 & Annual    & 500 m & 2012--present \\
    \bottomrule
  \end{tabular}
  \end{center}
  \vspace{0.3cm}
  \item Free access via NASA LAADS DAAC:
        \href{https://ladsweb.modaps.eosdis.nasa.gov}{\texttt{ladsweb.modaps.eosdis.nasa.gov}}
\end{itemize}
\end{frame}


\begin{frame}{Black Marble vs.\ DMSP-OLS}
\begin{center}
\small
\begin{tabular}{lcc}
  \toprule
  & \textbf{DMSP-OLS} & \textbf{NASA Black Marble (VIIRS)} \\
  \midrule
  Sensor        & DMSP / OLS      & Suomi-NPP VIIRS DNB \\
  Coverage      & 1992–2013       & 2012–present \\
  Resolution    & $\sim$1 km      & 500 m \\
  Radiometric   & 6-bit (0–63)    & 14-bit, calibrated nW/sr/cm$^2$ \\
  Saturation    & Yes (urban)     & No \\
  Moonlight corr.& No             & Yes \\
  Cloud mask    & Partial         & Yes \\
  Stray light corr.& No           & Yes \\
  Recommended for& Pre-2012, long run & 2012–present \\
  \bottomrule
\end{tabular}
\end{center}
\vspace{0.2cm}
\begin{itemize}
  \item \textbf{Key advantage of Black Marble:} fully calibrated, no saturation in
        bright urban areas --- suitable for studying within-country inequality
\end{itemize}
\end{frame}


\begin{frame}{BlackMarbleR Package}
\begin{itemize}
  \item \texttt{BlackMarbleR} is an R package that automates the full download
        pipeline: it identifies relevant HDF5 tiles from NASA LAADS DAAC, downloads
        them, mosaics the tiles, and converts to a \texttt{terra::SpatRaster}
  \vspace{0.3cm}
  \item GitHub: \href{https://worldbank.github.io/blackmarbler/}{\texttt{worldbank.github.io/blackmarbler}}
  \vspace{0.3cm}
  \item \textbf{Main function:}
  \vspace{0.1cm}
  \begin{itemize}
    \item \texttt{bm\_raster(roi\_sf, product\_id, date, bearer, ...)}
    \vspace{0.1cm}
    \item \texttt{roi\_sf}: region of interest as an \texttt{sf} polygon
    \vspace{0.1cm}
    \item \texttt{product\_id}: e.g.\ \texttt{"VNP46A4"} for annual composite
    \vspace{0.1cm}
    \item \texttt{bearer}: your NASA Earthdata Bearer token
  \end{itemize}
  \vspace{0.3cm}
  \item Returns a \texttt{SpatRaster} cropped and masked to your region
\end{itemize}
\end{frame}


\begin{frame}{Getting a NASA Earthdata Account}
\begin{enumerate}
  \item Register (free) at \href{https://urs.earthdata.nasa.gov}{\texttt{urs.earthdata.nasa.gov}}
  \vspace{0.25cm}
  \item Log in $\to$ Profile $\to$ \textbf{Generate Token}
  \vspace{0.25cm}
  \item Copy the token string (starts with \texttt{eyJ...}) and paste into the R script
  \vspace{0.25cm}
  \item Token expires after 1 year --- regenerate when needed
\end{enumerate}
\vspace{0.4cm}

\end{frame}

\begin{frame}{Where to Get Taiwan Boundaries}
\begin{itemize}
  \item \textbf{Option 1: GADM} (recommended for research reproducibility)
  \vspace{0.1cm}
  \begin{itemize}
    \item Global Administrative Areas: \href{https://gadm.org}{\texttt{gadm.org}}
    \vspace{0.1cm}
    \item Download directly in R with \texttt{geodata::gadm()}
    \vspace{0.1cm}
    \item Level 1 = counties / municipalities (22 units)\\
          Level 2 = townships / districts (鄉鎮市區, $\approx$368 units)
  \end{itemize}
  \vspace{0.3cm}
  \item \textbf{Option 2: Taiwan Government Open Data}
  \vspace{0.1cm}
  \begin{itemize}
    \item \href{https://data.gov.tw/dataset/7442}{\texttt{data.gov.tw}} --- 直轄市、縣市界線 / 鄉鎮市區界線
    \vspace{0.1cm}
    \item Official TWD97 projection (EPSG:3826); convert to WGS84 (EPSG:4326)
          before passing to BlackMarbleR
  \end{itemize}
  \vspace{0.3cm}
  \item \textbf{Option 3: \texttt{rnaturalearth} package}
  \vspace{0.1cm}
  \begin{itemize}
    \item Only provides country-level or province-level outlines
    \item Sufficient for a national overview map
  \end{itemize}
\end{itemize}
\end{frame}


\begin{frame}[fragile]{Downloading Taiwan Shapefile in R}
\begin{lstlisting}
library(pacman)
p_load(geodata, sf)

# Download GADM Level-2 (townships / districts)
tw_gadm <- geodata::gadm(
  country = "TWN",
  level   = 2,
  path    = rawdata_path   # cached locally
)
tw_sf <- sf::st_as_sf(tw_gadm)

# BlackMarbleR requires WGS84 (EPSG:4326)
tw_sf <- sf::st_transform(tw_sf, crs = 4326)

cat("Number of townships:", nrow(tw_sf))   # ~368
\end{lstlisting}
\begin{itemize}
  \item The shapefile is cached locally --- the download runs only once
  \vspace{0.2cm}
  \item Key columns: \texttt{NAME\_1} (county), \texttt{NAME\_2} (township),
        \texttt{HASC\_2} (ISO code)
\end{itemize}
\end{frame}

\begin{frame}[fragile]{Step 2: Download Nighttime Lights}
\begin{lstlisting}
# Annual composite for 2012 (first available year)
ntl_2012 <- bm_raster(
  roi_sf     = tw_sf,
  product_id = "VNP46A4",
  date       = "2012-01-01",  # any date in target year
  bearer     = bearer_token,
  output_location_type = "file",
  file_dir   = file.path(rawdata_path, "nightlights/")
)

# Annual composite for 2024 (most recent complete year)
ntl_2024 <- bm_raster(
  roi_sf     = tw_sf,
  product_id = "VNP46A4",
  date       = "2024-01-01",
  bearer     = bearer_token,
  output_location_type = "file",
  file_dir   = file.path(rawdata_path, "nightlights/")
)
\end{lstlisting}
\begin{itemize}
  \item \texttt{output\_location\_type = "file"} saves HDF5 tiles locally
        --- subsequent runs skip re-downloading
\end{itemize}
\end{frame}


\begin{frame}[fragile]{Step 3: Create Maps}
\begin{lstlisting}
# Convert raster to data frame for ggplot2
raster_to_df <- function(r, varname = "ntl") {
  df <- terra::as.data.frame(r, xy = TRUE)
  names(df) <- c("x", "y", varname)
  df[!is.na(df[[varname]]), ]
}

df_2012 <- raster_to_df(ntl_2012)
df_2024 <- raster_to_df(ntl_2024)

# Map function
make_ntl_map <- function(df, title_text) {
  ggplot(df, aes(x = x, y = y, fill = ntl)) +
    geom_raster() +
    geom_sf(data = tw_sf, fill = NA, color = "gray60",
            linewidth = 0.15, inherit.aes = FALSE) +
    scale_fill_viridis_c(option = "inferno",
      name = "Radiance\n(nW/sr/cm2)") +
    coord_sf(expand = FALSE) +
    labs(title = title_text, x = NULL, y = NULL) +
    theme_void()
}
\end{lstlisting}
\end{frame}


\begin{frame}[fragile]{Step 4: Aggregate to Township Level}
\begin{lstlisting}
# exact_extract() weights pixels by polygon coverage fraction
tw_ntl <- tw_sf %>%
  mutate(
    ntl_mean_2012 = exactextractr::exact_extract(
                      ntl_2012, tw_sf, "mean"),
    ntl_mean_2024 = exactextractr::exact_extract(
                      ntl_2024, tw_sf, "mean"),
    ntl_sum_2012  = exactextractr::exact_extract(
                      ntl_2012, tw_sf, "sum"),
    ntl_sum_2024  = exactextractr::exact_extract(
                      ntl_2024, tw_sf, "sum"),
    ntl_change    = ntl_mean_2024 - ntl_mean_2012,
    ntl_pct_change = (ntl_mean_2024 - ntl_mean_2012) /
                     (ntl_mean_2012 + 1) * 100
  )
\end{lstlisting}
\begin{itemize}
  \item We add 1 in the denominator to avoid division by zero for
        very dark areas ($\bar{Y}_{2012} \approx 0$)
\end{itemize}
\end{frame}
