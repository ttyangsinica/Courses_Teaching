\begin{frame}{Overview}{Division of Labor}
	\begin{itemize}
		\item Each team presents for 30-40 minutes, and every member must report on the part they are responsible for:
															\vspace{0.2cm}
		\begin{itemize}
			\item Member 1 reports on the Introduction and Data Construction.
																\vspace{0.2cm}
			\item Member 2 reports on Empirical Methods.
																\vspace{0.2cm}
			\item Member 3 reports and interprets the first part of the empirical results.
																\vspace{0.2cm}
			\item Member 4 reports and interprets the second part of the empirical results and conclusion.
																\vspace{0.2cm}
		\end{itemize}
		\item Please keep the number of slides to less than 30.
	\end{itemize}
\end{frame}

\begin{frame}{Content: Member 1}{Introduction}
	\begin{itemize}
		\item Try to answer the following questions:
															\vspace{0.2cm}
		\begin{itemize}
			\item What is the question? What causal relationship are the authors attempting to examine?
																\vspace{0.2cm}
			\item Why should we care about it?
																\vspace{0.2cm}
			\item How do the authors answer the research questions they propose?
			\begin{itemize}
																	\vspace{0.2cm}
				\item Briefly describe how the authors conduct the study and their findings.
			\end{itemize}
		\end{itemize}
	\end{itemize}
\end{frame}

\begin{frame}{Content: Member 1}{Data and Sample}
	\begin{itemize}
		\item Try to answer the following questions:
															\vspace{0.2cm}
		\begin{itemize}
			\item What is their data?
																\vspace{0.2cm}
			\begin{itemize}
				\item Briefly describe the main variables in the data and why this data is suitable for the study.
																	\vspace{0.2cm}
			\end{itemize}
			\item How the authors construct the sample for estimation.
																\vspace{0.2cm}
			\begin{itemize}
				\item Briefly describe the authors' sample construction and screening process.
																	\vspace{0.2cm}
				\item Discuss the descriptive statistics of the estimation sample.
			\end{itemize}
		\end{itemize}
	\end{itemize}
\end{frame}

\begin{frame}{Content: Member 2}{Empirical Specifications}
	\begin{itemize}
		\item Try to answer the following questions:
															\vspace{0.2cm}
		\begin{itemize}
			\item What is their empirical strategy?
																\vspace{0.2cm}
			\begin{itemize}
				\item Briefly describe the empirical methods used by the authors.
																	\vspace{0.2cm}
				\item Introduce in detail the estimation equations.
																	\vspace{0.2cm}
				\item What selection bias are the authors facing in estimating the causal relationship?
																	\vspace{0.2cm}
				\item Why does this method obtain the causal relationship the authors want to estimate, based on the underlying assumptions?
			\end{itemize}
		\end{itemize}
	\end{itemize}
\end{frame}

\begin{frame}{Content: Member 3}{Results}
	\begin{itemize}
		\item Try to answer the following questions:
															\vspace{0.2cm}
		\begin{itemize}
			\item What are their main results?
																\vspace{0.2cm}
			\begin{itemize}
				\item Interpret the authors' results in detail (graphs and tables).
																	\vspace{0.2cm}
				\item Is the estimated effect significant? How large is it?
																	\vspace{0.2cm}
			\end{itemize}
			\item Are their results robust to other settings?
																\vspace{0.2cm}
			\begin{itemize}
				\item What robustness checks have the authors conducted?
																	\vspace{0.2cm}
				\item Briefly describe whether the results are robust.
			\end{itemize}
		\end{itemize}
	\end{itemize}
\end{frame}

\begin{frame}{Content: Member 4}{Other Results and Conclusion}
	\begin{itemize}
		\item Try to answer the following questions:
															\vspace{0.2cm}
		\begin{itemize}
			\item Is there heterogeneity in the estimated effects?
																\vspace{0.2cm}
			\begin{itemize}
				\item Describe the subgroup analysis conducted by the authors.
																	\vspace{0.2cm}
				\item Are the estimated effects heterogeneous?
																	\vspace{0.2cm}
				\item How do the authors interpret these results?
																	\vspace{0.2cm}
			\end{itemize}
			\item Do they have any other interesting results?
																\vspace{0.2cm}
			\begin{itemize}
				\item Describe any other findings you find interesting.
																	\vspace{0.2cm}
				\item What further analysis do you think the authors could conduct?
																	\vspace{0.2cm}
			\end{itemize}
		\end{itemize}
		\item Briefly summarize the article's conclusions.
	\end{itemize}
\end{frame}
